' ===== UserForm1.frm =====
Attribute VB_Name = "UserForm1"
Attribute VB_Base = "0{C8427C69-6914-4433-969E-6B6AA476A467}{9B79BECF-2A02-4AE7-BDE2-948C6F0DB0B2}"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = False
Attribute VB_TemplateDerived = False
Attribute VB_Customizable = False
Private Sub UserForm_Initialize()

    ' --- Llenar Tipo de movimiento ---
    cboTipo.Clear
    cboTipo.AddItem "Entrada"
    cboTipo.AddItem "Salida"

    ' --- Llenar Solicitantes desde la hoja ---
    Dim ultimaFilaSolic As Long
    Dim i As Long
    
    ultimaFilaSolic = Sheets("Solicitantes").Cells(Rows.Count, 1).End(xlUp).Row
    
    cboSolicitante.Clear
    For i = 2 To ultimaFilaSolic   ' Empieza en fila 2 para saltar encabezados
        cboSolicitante.AddItem Sheets("Solicitantes").Cells(i, 1).Value & " - " & Sheets("Solicitantes").Cells(i, 2).Value
        ' Ejemplo: S001 - Juan Pérez
    Next i

    ' --- Llenar Equipos desde la hoja ---
    Dim ultimaFilaEq As Long
    
    ultimaFilaEq = Sheets("Inventario").Cells(Rows.Count, 1).End(xlUp).Row
    
    cboEquipo.Clear
    For i = 2 To ultimaFilaEq
        cboEquipo.AddItem Sheets("Inventario").Cells(i, 1).Value & " - " & Sheets("Inventario").Cells(i, 2).Value
        ' Ejemplo: EQ001 - Taladro Bosch
    Next i

End Sub

Private Sub cmdCerrar_Click()
Unload Me
End Sub

Private Sub cmdGuardar_Click()
 Dim fila As Long
    Dim filaInv As Long
    Dim codigoEq As String
    Dim cantMov As Long
    
    ' Buscar la primera fila vacía en la hoja Movimientos
    fila = Sheets("Movimientos").Cells(Rows.Count, 1).End(xlUp).Row + 1
    
    ' Insertar datos en la hoja Movimientos
    With Sheets("Movimientos")
        .Cells(fila, 1).Value = "M" & Format(fila, "000") ' ID Movimiento automático
        .Cells(fila, 2).Value = txtFecha.Value
        .Cells(fila, 3).Value = cboTipo.Value
        .Cells(fila, 4).Value = cboSolicitante.Value
        .Cells(fila, 5).Value = cboEquipo.Value
        .Cells(fila, 6).Value = txtCantidad.Value
        .Cells(fila, 7).Value = txtObs.Value
    End With
    
    ' --- Actualizar Inventario ---
    codigoEq = Split(cboEquipo.Value, " - ")(0) ' Extrae el código (ej: EQ001)
    cantMov = CLng(txtCantidad.Value)
    
    ' Buscar el equipo en la hoja Inventario
    filaInv = Application.Match(codigoEq, Sheets("Inventario").Range("A:A"), 0)
    
    If Not IsError(filaInv) Then
        With Sheets("Inventario")
            If cboTipo.Value = "Salida" Then
                .Cells(filaInv, 4).Value = .Cells(filaInv, 4).Value - cantMov
            ElseIf cboTipo.Value = "Entrada" Then
                .Cells(filaInv, 4).Value = .Cells(filaInv, 4).Value + cantMov
            End If
        End With
    End If
    
    MsgBox "Movimiento registrado y stock actualizado correctamente", vbInformation
End Sub

Private Sub cmdNuevo_Click()
txtFecha.Value = ""
    cboTipo.Value = ""
    cboSolicitante.Value = ""
    cboEquipo.Value = ""
    txtCantidad.Value = ""
    txtObs.Value = ""
End Sub
